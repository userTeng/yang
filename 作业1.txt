#include <iostream>
#include <stdlib.h>
#include <time.h>
#include <cstdlib>

using namespace std;

int main(int argc, char* argv[])
{
	int x = atoi(argv[1]);
    srand((unsigned)time(NULL));
    for(int i=0; i<x;i++)
    {
         cout << rand() << "\n";
    }
    cout << endl;

    return 0;
}
