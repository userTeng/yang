const int w_1m=1000000;
int w[w_1m];

bool is_match(int t, int w[], int w_length)
{
    /*for(int i=0;i<w_length;i++)
    {
        if(t!=w[i])
        {
            return true;
        }
    }*/
    int left =0;
    int rigrht=w_length-1;
    int mid=0;
    while(left<=rigrht){
    	mid=(left+rigrht)/2;
    	if(t==w[mid]){
    		return false;
    	}
    	else if(t<w[mid]){
    		rigrht=mid-1;
    	}
    	else if(t>w[mid]){
    		left=mid+1;
    	}
    }
    return true;
}
